            addAttribute(PositionIncrementAttribute.class);

    private int levelsSkipped = 0;
    private boolean pendingDelimiter = false;
    private StringBuilder currentPathBuffer;

    private int inputCharsRead = 0;

    @Override
    public final boolean incrementToken() throws IOException {
        clearAttributes();
        termAttribute.append(currentPathBuffer);
        positionIncrementAttribute.setPositionIncrement(1);
        int length = 0;
        boolean added = false;
        if (pendingDelimiter) {
            termAttribute.append(replacement);
            length++;
            pendingDelimiter = false;
            added = true;
        }

        while (true) {
            int c = input.read();
            if (c >= 0) {
                inputCharsRead++;
            } else {
                if (levelsSkipped > skip) {
                    length += currentPathBuffer.length();
                    termAttribute.setLength(length);
                    offsetAttribute.setOffset(correctOffset(pathStartPos), correctOffset(pathStartPos + length));
                    if (added) {
                        currentPathBuffer.setLength(0);
                        currentPathBuffer.append(termAttribute.buffer(), 0, length);
                    }
                    return added;
                } else {
                    return false;
                }
            }
            if (!added) {
                added = true;
                levelsSkipped++;
                if (levelsSkipped > skip) {
                    termAttribute.append(c == delimiter ? replacement : (char) c);
                    length++;
                } else {
                    pathStartPos++;
                }
            } else {

                        break;

                    levelsSkipped++;

        termAttribute.setLength(length);
        offsetAttribute.setOffset(correctOffset(pathStartPos), correctOffset(pathStartPos + length));

        currentPathBuffer.append(termAttribute.buffer(), 0, length);
        return true;

        currentPathBuffer.setLength(0);
        inputCharsRead = 0;
        pendingDelimiter = false;
        levelsSkipped = 0;
        pathStartPos = 0;
    }
