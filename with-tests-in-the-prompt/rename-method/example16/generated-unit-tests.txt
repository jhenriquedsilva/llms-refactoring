package org.springframework.core.annotation;

import org.junit.*;
import org.junit.rules.ExpectedException;

import java.lang.annotation.*;
import java.lang.reflect.Method;

import static org.junit.Assert.*;

public class AttributeMethodsTest {

    // Test annotation with various attribute types
    @Retention(RetentionPolicy.RUNTIME)
    private @interface SampleAnnotation {
        String value();
        int number() default 42;
        NestedAnnotation nested() default @NestedAnnotation;
        Class<?> clazz() default Object.class;
        Day day() default Day.SUNDAY;
    }
    @Retention(RetentionPolicy.RUNTIME)
    private @interface NestedAnnotation {
        String nestedValue() default "nested";
    }
    private enum Day { MONDAY, SUNDAY }

    @Retention(RetentionPolicy.RUNTIME)
    private @interface NoAttributes {}

    // Helper to get AttributeMethods for a given annotation
    private static AttributeMethods getMethods(Class<? extends Annotation> type) {
        return AttributeMethods.forAnnotationType(type);
    }

    // ------------- isValid(Annotation) ------------------

    @Test
    public void testIsValidTypical() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        SampleAnnotation ann = new SampleAnnotation() {
            public String value() { return "abc"; }
            public int number() { return 77; }
            public NestedAnnotation nested() { return new NestedAnnotation() {
                public String nestedValue() { return "N"; }
                public Class<? extends Annotation> annotationType() { return NestedAnnotation.class; }
            }; }
            public Class<?> clazz() { return String.class; }
            public Day day() { return Day.MONDAY; }
            public Class<? extends Annotation> annotationType() { return SampleAnnotation.class; }
        };
        assertTrue(am.isValid(ann));
    }

    @Test
    public void testIsValidWithNullAnnotation() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        try {
            am.isValid(null);
            fail("Expected IllegalArgumentException or AssertionError");
        } catch (Exception ex) {
            // expected
        }
    }

    @Test
    public void testIsValidWithWrongTypeAnnotation() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        NoAttributes na = new NoAttributes() {
            public Class<? extends Annotation> annotationType() {
                return NoAttributes.class;
            }
        };
        try {
            am.isValid(na);
            fail("Expected IllegalArgumentException or AssertionError");
        } catch (Exception ex) {
            // expected
        }
    }

    // ------------ validate(Annotation) ------------------

    @Rule
    public ExpectedException thrown = ExpectedException.none();

    @Test
    public void testValidateTypical() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        SampleAnnotation ann = new SampleAnnotation() {
            public String value() { return "abc"; }
            public int number() { return 77; }
            public NestedAnnotation nested() { return new NestedAnnotation() {
                public String nestedValue() { return "N"; }
                public Class<? extends Annotation> annotationType() { return NestedAnnotation.class; }
            }; }
            public Class<?> clazz() { return String.class; }
            public Day day() { return Day.SUNDAY; }
            public Class<? extends Annotation> annotationType() { return SampleAnnotation.class; }
        };
        am.validate(ann); // should pass
    }

    @Test
    public void testValidateNullAnnotation() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        thrown.expect(IllegalArgumentException.class);
        am.validate(null);
    }

    @Test
    public void testValidateWrongTypeAnnotation() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        NoAttributes na = new NoAttributes() {
            public Class<? extends Annotation> annotationType() {
                return NoAttributes.class;
            }
        };
        thrown.expect(IllegalArgumentException.class);
        am.validate(na);

    // ------------ get(String name) ------------------

    @Test
    public void testGetByNameFound() {
        AttributeMethods am = getMethods(SampleAnnotation.class);
        Method method = am.get("value");
        assertNotNull(method);
        assertEquals("value", method.getName());
    }

    @Test
    public void testGetByNameNotFound() {

    }

    @Test
    public void testDescribeClassAndNameTypical() {

    }

    @Test
    public void testDescribeClassAndNameNull() {
        String desc = AttributeMethods.describe(null, null);
        assertEquals("(none)", desc);
    }

    @Test
    public void testDescribeClassNullNamePresent() {

        assertEquals("attribute 'testName'", desc);
    }
