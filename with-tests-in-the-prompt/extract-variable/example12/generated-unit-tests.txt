package org.eclipse.jgit.internal.storage.commitgraph;

import org.eclipse.jgit.errors.CorruptObjectException;
import org.eclipse.jgit.errors.IncorrectObjectTypeException;
import org.eclipse.jgit.errors.MissingObjectException;
import org.eclipse.jgit.internal.storage.io.CancellableDigestOutputStream;
import org.eclipse.jgit.lib.ObjectId;
import org.eclipse.jgit.lib.ObjectReader;
import org.eclipse.jgit.lib.ProgressMonitor;
import org.eclipse.jgit.revwalk.RevCommit;
import org.junit.Before;
import org.junit.Test;

import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.InterruptedIOException;
import java.io.OutputStream;
import java.util.Collections;
import java.util.List;

import static org.junit.Assert.*;
import static org.mockito.Mockito.*;

public class CommitGraphWriterTest {

    private GraphCommits emptyGraphCommits;
    private GraphCommits singleGraphCommits;
    private ProgressMonitor mockMonitor;

    @Before
    public void setup() {
        emptyGraphCommits = mock(GraphCommits.class);
        when(emptyGraphCommits.size()).thenReturn(0);
        when(emptyGraphCommits.iterator()).thenReturn(Collections.emptyIterator());

        // singleGraphCommits will simulate one commit
        singleGraphCommits = mock(GraphCommits.class);
        RevCommit mockCommit = mock(RevCommit.class);
        when(mockCommit.getFirstByte()).thenReturn((byte)0x01);
        when(mockCommit.getCommitTime()).thenReturn(123456);
        when(singleGraphCommits.size()).thenReturn(1);
        when(singleGraphCommits.iterator()).thenReturn(Collections.singleton(mockCommit).iterator());
        when(singleGraphCommits.getExtraEdgeCnt()).thenReturn(0);
        mockMonitor = mock(ProgressMonitor.class);
    }

    // Test constructor with only graphCommits
    @Test
    public void testConstructor_singleArg() {
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits);
        assertNotNull(writer);
    }

    // Test constructor with generateChangedPathFilters true
    @Test
    public void testConstructor_bothArgs() {
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, true);
        assertNotNull(writer);
    }

    // Test write with an empty graph returns Stats.EMPTY
    @Test
    public void testWrite_emptyGraph() throws IOException {
        CommitGraphWriter writer = new CommitGraphWriter(emptyGraphCommits, false);
        OutputStream out = new ByteArrayOutputStream();
        CommitGraphWriter.Stats stats = writer.write(mockMonitor, out);
        assertSame(CommitGraphWriter.Stats.EMPTY, stats);
    }

    // Test basic write with a single commit; does not throw exception
    @Test
    public void testWrite_singleCommit() throws IOException {
        // Simulate all item methods
        RevCommit mockCommit = mock(RevCommit.class);
        when(mockCommit.getFirstByte()).thenReturn((byte)0x01);
        when(mockCommit.getCommitTime()).thenReturn(123456);
        when(mockCommit.getParents()).thenReturn(new RevCommit[0]);
        when(mockCommit.getTree()).thenReturn(mock(ObjectId.class));
        when(singleGraphCommits.iterator()).thenReturn(Collections.singleton(mockCommit).iterator());
        when(singleGraphCommits.size()).thenReturn(1);
        when(singleGraphCommits.getExtraEdgeCnt()).thenReturn(0);
        when(singleGraphCommits.getOidPosition(any())).thenReturn(0);

        // Create ProgressMonitor mock
        ProgressMonitor monitor = mock(ProgressMonitor.class);

        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, false);
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        CommitGraphWriter.Stats stats = writer.write(monitor, baos);
        assertNotNull(stats);
        assertEquals(0, stats.getChangedPathFiltersReused());
        assertEquals(0, stats.getChangedPathFiltersComputed());
        // Should generate output bytes
        assertTrue(baos.size() > 0);
    }

    // Test write() throws IOException when OutputStream throws
    @Test(expected = IOException.class)
    public void testWrite_outputStreamThrows() throws IOException {
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, false);
        OutputStream badStream = mock(OutputStream.class);
        when(badStream.write(any(byte[].class), anyInt(), anyInt())).thenThrow(new IOException("fail"));
        writer.write(mockMonitor, badStream);
    }

    // Test write() throws IOException when interrupted (simulated)
    @Test(expected = IOException.class)
    public void testWrite_interruptedIOException() throws IOException {
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, false);
        OutputStream badStream = mock(OutputStream.class);
        doThrow(new InterruptedIOException("interrupted")).when(badStream).write(any(byte[].class), anyInt(), anyInt());
        writer.write(mockMonitor, badStream);
    }

    // Test edge case: missing monitor (should not throw NPE, because annotated as NonNull)
    @Test(expected = NullPointerException.class)
    public void testWrite_nullMonitor() throws IOException {
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, false);
        OutputStream out = new ByteArrayOutputStream();
        writer.write(null, out);
    }

    // Test edge case: missing OutputStream
    @Test(expected = NullPointerException.class)
    public void testWrite_nullOutStream() throws IOException {
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, false);
        writer.write(mockMonitor, null);
    }

    // Test edge case: generation of BloomFilterChunks (generateChangedPathFilters = true)
    @Test
    public void testWrite_bloomFilterEnabled() throws Exception {
        // Set up GraphCommits to simulate no changed path filters reused
        RevCommit mockCommit = mock(RevCommit.class);
        when(mockCommit.getFirstByte()).thenReturn((byte) 0x03);
        when(mockCommit.getCommitTime()).thenReturn(42);
        when(mockCommit.getParents()).thenReturn(new RevCommit[0]);
        when(mockCommit.getTree()).thenReturn(mock(ObjectId.class));
        when(mockCommit.getChangedPathFilter(any())).thenReturn(null); // So that compute is always called

        GraphCommits graphCommits = mock(GraphCommits.class);
        when(graphCommits.size()).thenReturn(1);
        when(graphCommits.iterator()).thenReturn(Collections.singleton(mockCommit).iterator());
        when(graphCommits.getExtraEdgeCnt()).thenReturn(0);
        when(graphCommits.getOidPosition(any())).thenReturn(0);
        when(graphCommits.getObjectReader()).thenReturn(mock(ObjectReader.class));

        CommitGraphWriter writer = new CommitGraphWriter(graphCommits, true);
        ProgressMonitor monitor = mock(ProgressMonitor.class);
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        CommitGraphWriter.Stats stats = writer.write(monitor, baos);

        assertNotNull(stats);
        // At least one filter should be computed
        assertTrue(stats.getChangedPathFiltersComputed() >= 1);
        // Reused filters might be zero
        assertTrue(stats.getChangedPathFiltersReused() >= 0);
    }

    // Test to ensure IllegalStateException is thrown if expected output size doesn't match
    @Test(expected = IllegalStateException.class)
    public void testWrite_illegalStateWhenChecksumMismatch() throws Exception {
        // Patch out.length() to deliberately mismatch
        CommitGraphWriter writer = new CommitGraphWriter(singleGraphCommits, false);
        CancellableDigestOutputStream badOut = mock(CancellableDigestOutputStream.class);
        when(badOut.length()).thenReturn(100L); // actual
        // prepare so writer.writeHeader() works
        doNothing().when(badOut).write(any(byte[].class), anyInt(), anyInt());
        doNothing().when(badOut).flush();
        when(badOut.getWriteMonitor()).thenReturn(mockMonitor);
        doNothing().when(badOut).write(any(byte[].class));
        when(badOut.getDigest()).thenReturn(new byte[20]);

        OutputStream os = spy(new ByteArrayOutputStream());
        // Trick so out.length() returns a mismatch (simulate failure after)
        CommitGraphWriter tester = new CommitGraphWriter(singleGraphCommits, false) {
            @Override
            public Stats write(ProgressMonitor monitor, OutputStream commitGraphStream) throws IOException {
                // Skip to the exception point directly
                List fakeChunks = Collections.emptyList();
                long expectedSize = 200L;
                // This should fail due to expectedSize != out.length()
                try (CancellableDigestOutputStream out = badOut) {
                    writeHeader(out, 0);
                    writeChunkLookup(out, fakeChunks);
                    writeChunks(out, fakeChunks);
                    writeCheckSum(out);
                    if (expectedSize != out.length()) {
                        throw new IllegalStateException("commitGraphUnexpectedSize");
                    }
                }
                return CommitGraphWriter.Stats.EMPTY;
            }
        };
        tester.write(mockMonitor, os);
    }

    // ===== Tests for CommitGraphWriter.Stats accessors =====

    @Test
    public void testEmptyStatsGetters() {
        CommitGraphWriter.Stats stats = CommitGraphWriter.Stats.EMPTY;
        assertEquals(0, stats.getChangedPathFiltersReused());
        assertEquals(0, stats.getChangedPathFiltersComputed());
    }

    @Test
    public void testStatsFromBloomFilterChunks() throws Exception {
        // Simulate a Stats created from a BloomFilterChunks with counts
        ByteArrayOutputStream idx = new ByteArrayOutputStream();
        ByteArrayOutputStream dat = new ByteArrayOutputStream();
        long reused = 5L, computed = 3L;
        // Instantiate via reflection (inner class, so build it via the parent)
        Object bfChunks = CommitGraphWriter.class
            .getDeclaredClasses()[1]
            .getConstructor(ByteArrayOutputStream.class, ByteArrayOutputStream.class, long.class, long.class)
            .newInstance(idx, dat, reused, computed);

        CommitGraphWriter.Stats stats = CommitGraphWriter.Stats.from((CommitGraphWriter.BloomFilterChunks) bfChunks);
        assertEquals(computed, stats.getChangedPathFiltersComputed());
        assertEquals(reused, stats.getChangedPathFiltersReused());
    }
}
